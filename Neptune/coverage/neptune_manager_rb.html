714   def acquire_nodes(job_data)
 
715     # for jobs where no nodes need to be acquired (e.g., concurrent but not
 
716     # distributed programs), run them on the shadow node
 
717     if NO_NODES_NEEDED.include?(job_data["@type"])
 
718       NeptuneManager.log("No nodes needed for job type [#{job_data['@type']}]," +
 
719         " not acquiring nodes")
 
720       return [my_node()]
 
721     end
 
722 
723     NeptuneManager.log("acquiring nodes")
 
724 
725     #num_of_vms_needed = optimal_nodes_hill_climbing(job_data, "performance")
 
726     nodes_needed = optimal_nodes(job_data)
 
727 
728     NeptuneManager.log("acquiring nodes for hybrid cloud neptune job")
 
729 
730     if nodes_needed.class == Array
 
731       nodes_needed = Hash[*nodes_needed]
 
732       NeptuneManager.log("request received to spawn hybrid nodes: #{nodes_needed.inspect}")
 
733     elsif nodes_needed.class == Fixnum
 
734       nodes_needed = {"cloud1" => nodes_needed}
 
735     else
 
736       NeptuneManager.log("nodes_needed was not the right class - should have been Array or Fixnum but was #{nodes_needed.class}")
 
737       # TODO: find a way to reject the job here
 
738     end
 
739 
740     nodes_to_use = []
 
741 
742     nodes_needed.each { |cloud, nodes_to_acquire|
 
743       # nodes_to_acquire can either be an integer or a URL
 
744       # if it's an integer, spawn up that many nodes
 
745       # if it's a URL, it refers to a remote cloud resource we don't control
 
746       # (e.g., Google App Engine), so skip it
 
747 
748       # in non-hybrid cloud runs, cloud1 will be the only cloud that specifies
 
749       # an integer value
 
750       if nodes_to_acquire =~ URL_REGEX
 
751         NeptuneManager.log("nodes to acquire for #{cloud} was a URL " + 
752           "[#{nodes_to_acquire}], so not spawning nodes")
 
753         next
 
754       end
 
755 
756       NeptuneManager.log("acquiring #{nodes_to_acquire} nodes for #{cloud}")
 
757       nodes_for_cloud = find_open_nodes(cloud, nodes_to_acquire, job_data)
 
758       nodes_to_use = [nodes_to_use + nodes_for_cloud].flatten
 
759       # TODO: should check for failures acquiring nodes
 
760     }
 
761 
762     return nodes_to_use
 
763   end

888 =begin
 
889   Hill Climbing Algorithm
 
890     # find minimum execution time t1
 
891     # find neighbors t0 and t2
 
892 
893     # if t0 is too low set it to t1
 
894     # if t2 is too high set it to t1
 
895 
896     # if no data for either, choose t2
 
897     # if no data for t0, choose t0
 
898     # if data for both, choose t1

900   def optimal_nodes_hill_climbing(job_data, thing_to_optimize)
 
901     job_name = get_job_name(job_data)
 
902 
903     if thing_to_optimize != "cost" and thing_to_optimize != "performance"
 
904       abort("bad thing to optimize - can be cost or performance but was #{thing_to_optimize}")
 
905     end
 
906 
907     current_data = @jobs[job_name]
 
908     if current_data.nil? or current_data.empty?
 
909       NeptuneManager.log("neptune - no job data yet for [#{job_name}]")
 
910       return job_data["@nodes_to_use"]
 
911     end
 
912 
913     NeptuneManager.log("found job data for [#{job_name}]")
 
914 
915     min_val = INFINITY
 
916     optimal_job = nil
 
917     current_data.each { |job|
 
918       NeptuneManager.log("current job data is [#{job}]")
 
919 
920       if thing_to_optimize == "performance"
 
921         my_val = job.total_time
 
922       elsif thing_to_optimize == "cost"
 
923         my_val = job.cost
 
924       else
 
925         abort("bad thing to optimize again")
 
926       end
 
927 
928       if my_val < min_val
 
929         NeptuneManager.log("found a new minimum - [#{job}]")
 
930         optimal_job = job
 
931       end
 
932     }
 
933 
934     NeptuneManager.log("minimum is - [#{optimal_job}]")
 
935 
936     search_space = job_data["@can_run_on"]
 
937     t1 = optimal_job.num_nodes
 
938 
939     NeptuneManager.log("optimal right now is t1 = #{t1}")
 
940     t0, t2 = find_neighbors(t1, search_space)
 
941     NeptuneManager.log("t1's neighbors are #{t0} and t2 = #{t2}")
 
942 
943     d0 = get_job_data(job_name, t0)
 
944     d2 = get_job_data(job_name, t2)
 
945 
946     return t2 if d0.nil? and d2.nil?
 
947     return t0 if d0.nil?
 
948     return t1
 
949   end

952   def find_neighbors(val, search_space)
 
953     abort("no empty arrays") if search_space.nil? or search_space.empty?
 
954 
955     left, right = nil, nil
 
956     length = search_space.length
 
957     search_space.each_with_index { |item, index|
 
958       # set left
 
959       if index <= 0
 
960         left = val
 
961       else
 
962         left = search_space[index-1]
 
963       end
 
964 
965       # set right
 
966       if index < length - 1
 
967         right = search_space[index+1]
 
968       else
 
969         right = val
 
970       end
 
971 
972       break if item == val
 
973     }
 
974 
975     return left, right
 
976   end
