151   # Given an array containing public and private IPs, separates them into
 
152   # one public IP array and a private IP array. Any private IPs are then
 
153   # checked to see if they resolve properly (which may fail in hybrid cloud
 
154   # deployments), and any private IPs that do not resolve are replaced with

156   def self.get_ips(ips)
 
157     abort("ips not even length array") if ips.length % 2 != 0
 
158     reported_public = []
 
159     reported_private = []
 
160     ips.each_index { |index|
 
161       if index % 2 == 0
 
162         reported_public << ips[index]
 
163       else
 
164         reported_private << ips[index]
 
165       end
 
166     }
 
167     
168     Kernel.puts("Reported Public IPs: [#{reported_public.join(', ')}]")
 
169     Kernel.puts("Reported Private IPs: [#{reported_private.join(', ')}]")
 
170 
171     actual_public = []
 
172     actual_private = []
 
173     
174     reported_public.each_index { |index|
 
175       pub = reported_public[index]
 
176       pri = reported_private[index]
 
177       if pub != "0.0.0.0" and pri != "0.0.0.0"
 
178         actual_public << pub
 
179         actual_private << pri
 
180       end
 
181     }
 
182         
183     actual_private.each_index { |index|
 
184       begin
 
185         actual_private[index] = HelperFunctions.convert_fqdn_to_ip(actual_private[index])
 
186       rescue Exception
 
187         # this can happen if the private ip doesn't resolve
 
188         # which can happen in hybrid environments: euca boxes wont be 
189         # able to resolve ec2 private ips, and vice-versa in euca-managed-mode
 
190         Kernel.puts("rescued! failed to convert #{actual_private[index]} to public")
 
191         actual_private[index] = actual_public[index]
 
192       end
 
193     }
 
194     
195     return actual_public, actual_private
 
196   end

454   # Creates a new security group in the given cloud infrastructure and opens
 
455   # all the TCP, UDP, and ICMP ports within it. We open the ports under the
 
456   # presumption that the AppScale firewall rules will be applied to iptables
 
457   # to lock back down any ports that need not be open to the world.
 
458   def self.create_appscale_security_group(infrastructure, group)
 
459     self.shell("#{infrastructure}-add-group #{group} -d appscale 2>&1")
 
460     self.shell("#{infrastructure}-authorize #{group} -p 1-65535 -P udp 2>&1")
 
461     self.shell("#{infrastructure}-authorize #{group} -p 1-65535 -P tcp 2>&1")
 
462     self.shell("#{infrastructure}-authorize #{group} -s 0.0.0.0/0 -P icmp -t -1:-1 2>&1")
 
463   end

473   # Obscures out sensitive strings by replacing most of the characters with
 
474   # a dummy character. Callers can use this method to print out credentials
 
475   # that may be logged and thus sent over e-mail (and thus should not be
 
476   # completely exposed in cleartext).
 
477   def self.obscure_string(string)
 
478     return string if string.nil? or string.length < 4
 
479     last_four = string[string.length-4, string.length]
 
480     obscured = "*" * (string.length-4)
 
481     return obscured + last_four
 
482   end
