969   # Starts the given roles by using open nodes, spawning new nodes, or some
 
970   # combination of the two. 'nodes_needed' should be an Array, where each 
971   # item is an Array of the roles to start on each node.
 
972   def start_new_roles_on_nodes(nodes_needed, instance_type, secret)
 
973      if !valid_secret?(secret)
 
974       return BAD_SECRET_MSG
 
975     end
 
976 
977     if nodes_needed.class != Array
 
978       Djinn.log_debug("Was expecting nodes_needed to be an Array, not " +
 
979         "a #{nodes_needed.class}")
 
980       return BAD_INPUT_MSG
 
981     end
 
982 
983     Djinn.log_debug("Received a request to acquire nodes with roles " +
 
984       "#{nodes_needed.join(', ')}, with instance type #{instance_type} for " +
 
985       "new nodes")
 
986 
987     vms_to_use = []
 
988     ZKInterface.lock_and_run {
 
989       num_of_vms_needed = nodes_needed.length
 
990 
991       @nodes.each { |node|
 
992         if node.is_open?
 
993           Djinn.log_debug("Will use node #{node} to run new roles")
 
994           vms_to_use << node
 
995 
996           if vms_to_use.length == nodes_needed.length
 
997             Djinn.log_debug("Only using open nodes to run new roles")
 
998             break
 
999           end
 
1000         end
 
1001       }
 
1002 
1003       vms_to_spawn = nodes_needed.length - vms_to_use.length
 
1004     
1005       if vms_to_spawn > 0 and !is_cloud?
 
1006         Djinn.log_debug("Still need #{vms_to_spawn} more nodes, but we " +
 
1007         "aren't in a cloud environment, so we can't acquire more nodes - " +
 
1008         "failing the caller's request.")
 
1009         return NOT_ENOUGH_OPEN_NODES
 
1010       end
 
1011 
1012       if vms_to_spawn > 0
 
1013         Djinn.log_debug("Need to spawn up #{vms_to_spawn} VMs")
 
1014         # start up vms_to_spawn vms as open
 
1015         imc = InfrastructureManagerClient.new(@@secret)
 
1016         new_nodes_info = imc.spawn_vms(vms_to_spawn, @creds, "open", "cloud1")
 
1017 
1018         # initialize them and wait for them to start up
 
1019         Djinn.log_debug("info about new nodes is " +
 
1020           "[#{new_nodes_info.join(', ')}]")
 
1021         add_nodes(new_nodes_info) 
1022 
1023         # add information about the VMs we spawned to our list, which may
 
1024         # already have info about the open nodes we want to use
 
1025         new_nodes = Djinn.convert_location_array_to_class(new_nodes_info,
 
1026           @creds['keyname'])
 
1027         vms_to_use << new_nodes
 
1028         vms_to_use.flatten!
 
1029       end
 
1030     }
 
1031 
1032     wait_for_nodes_to_finish_loading(vms_to_use)
 
1033     
1034     nodes_needed.each_index { |i|
 
1035       Djinn.log_debug("Adding roles #{nodes_needed[i].join(', ')} " +
 
1036         "to virtual machine #{vms_to_use[i]}")
 
1037       ZKInterface.add_roles_to_node(nodes_needed[i], vms_to_use[i])
 
1038     }
 
1039 
1040     wait_for_nodes_to_finish_loading(vms_to_use)
 
1041 
1042     return "OK"
 
1043   end
