183   # (between other AppControllers) to the ZooKeeper hierarchy located at

213   # This method provides callers with an easier way to read and write to
 
214   # AppController data in ZooKeeper. This is useful for methods that aren't
 
215   # sure if they already have the ZooKeeper lock or not, but definitely need
 
216   # it and don't want to accidentally cause a deadlock (grabbing the lock when
 
217   # they already have it).
 
218   def self.lock_and_run(&block)
 
219     # Create the ZK lock path if it doesn't exist.
 
220     if !self.exists?(APPCONTROLLER_PATH)
 
221       self.set(APPCONTROLLER_PATH, DUMMY_DATA, NOT_EPHEMERAL)
 
222     end
 
223 
224     # Try to get the lock, and if we can't get it, see if we already have
 
225     # it. If we do, move on (but don't release it later since this block
 
226     # didn't grab it), and if we don't have it, try again.
 
227     got_lock = false
 
228     begin
 
229       if self.get_appcontroller_lock()
 
230         got_lock = true
 
231       else  # it may be that we already have the lock
 
232         info = self.run_zookeeper_operation {
 
233           @@zk.get(:path => APPCONTROLLER_LOCK_PATH)
 
234         }
 
235         owner = JSON.load(info[:data])
 
236         if @@client_ip == owner
 
237           Djinn.log_debug("Tried to get the lock, but we already have it. " +
 
238             "Not grabbing/releasing lock again.")
 
239           got_lock = false
 
240         else 
241           Djinn.log_debug("Tried to get the lock, but it's currently owned " +
 
242             "by #{owner}. Will try again later.")
 
243           raise Exception
 
244         end
 
245       end
 
246     rescue Exception => e
 
247       Djinn.log_debug("Saw an exception of class #{e.class}")
 
248       Kernel.sleep(5)
 
249       retry
 
250     end
 
251 
252     begin
 
253       yield  # invoke the user's block, and catch any uncaught exceptions
 
254     rescue Exception => except
 
255       Djinn.log_debug("Ran caller's block but saw an Exception of class " +
 
256         "#{except.class}")
 
257       raise except
 
258     ensure
 
259       if got_lock
 
260         Djinn.log_debug("Grabbed lock, so now releasing it.")
 
261         self.release_appcontroller_lock()
 
262         Djinn.log_debug("Released lock successfully")
 
263       else
 
264         Djinn.log_debug("Didn't grab lock, so not releasing it.")
 
265       end
 
266     end
 
267   end
